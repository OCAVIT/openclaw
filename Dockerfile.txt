# Используем легкий образ Node.js
FROM node:20-slim

# 1. Устанавливаем Chromium и зависимости для его работы
# Это критически важный шаг для Railway/Linux
RUN apt-get update && apt-get install -y \
    chromium \
    fonts-ipafont-gothic fonts-wqy-zenhei fonts-thai-tlwg fonts-kacst fonts-freefont-ttf libxss1 \
    --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*

# 2. Настраиваем переменные окружения для Puppeteer
# Указываем, чтобы он не качал свой Chrome, а использовал системный
ENV PUPPETEER_SKIP_CHROMIUM_DOWNLOAD=true \
    PUPPETEER_EXECUTABLE_PATH=/usr/bin/chromium

# 3. Настройка рабочей директории
WORKDIR /app

# 4. Копируем package.json и устанавливаем зависимости
COPY package*.json ./
RUN npm install

# 5. Копируем остальной код
COPY . .

# 6. Билдим TypeScript (если проект на TS)
# Если в package.json нет скрипта build, закомментируй эту строку
RUN npm run build --if-present

# 7. Открываем порт (обычно агенты слушают 3000 или 8080)
EXPOSE 3000

# 8. Запускаем
CMD ["npm", "start"]
